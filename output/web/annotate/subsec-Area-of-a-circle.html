<subsection xml:id="subsec-Area-of-a-circle">
  <title>Area of a circle</title>
  <p>
    Learning point.
  </p>
  <image width="40%">
  <latex-image>
    \begin{tikzpicture}
      \draw[very thick,fill=white](0,0) circle(5);
      \draw[thick,fill=black](0,0) -- (5,0);
      \node [very thick]at (2.5,0.5){$r$};
      \node at (0,0){.};
    \end{tikzpicture}
  </latex-image>
    
  </image>
  <p>
    <ul>
      <li>
        <p>
          The area of a circle <me>= \pi r \times r</me>
        </p>
        <p>
          <me>=\pi r^2</me>
        </p>
        <p>
          where:
        </p>
        <p>
          <m>r</m> is the radius of the circle and <m>\pi=\frac{22}{7} \text{or} 3.142 </m>.
        </p>
      </li>
    </ul>
  </p>
  <activity>
    <statement>
      <p>
        Find the area of the circle below.
      </p>
      <image width="40%">
      <latex-image>
        \begin{tikzpicture}
          \draw[very thick,fill=green](0,0) circle(5);
          \draw[thick,fill=black](0,0) -- (5,0);
          \node [very thick]at (1,0.5){$14cm$};
          \node at (0,0){.};
        \end{tikzpicture}
      </latex-image>
        
      </image>
    </statement>
    <solution>
      <p>
        <md>
          <mrow> \text{Area of a circle}=\amp \pi r^2 </mrow>
          <mrow> =\amp\frac{22}{7} \times r \times r </mrow>
          <mrow> =\amp \frac{22}{7} \times 14 \times 14 cm^2 </mrow>
          <mrow> =\amp \frac{22}{7} \times 196 cm^2</mrow>
          <mrow> =\amp 616 cm^2</mrow>
        </md>
      </p>
    </solution>
  </activity>
  <example>
    <statement>
      <p>
        Calculate the area of a circle whose diameter is <m>10 cm</m> 
        <m>(\text{take} \pi=3.142)</m>
      </p>
    </statement>
    <solution>
      <p>
        <md>
          <mrow>\text{radius} =\amp \frac{\text{diameter}}{2}</mrow>
          <mrow> r= \amp \frac{10cm}{2}</mrow>
          <mrow> = \amp 5cm</mrow>
        </md>
      </p>
      <p>
        Therefore,
      </p>
      <p>
        <md>
          <mrow> \text{Area of a circle}=\amp 3.142 \times r \times r </mrow>
          <mrow> =\amp 3.142 \times5cm \times 5cm </mrow>
          <mrow> =\amp 78.5cm^2</mrow>
        </md>
      </p>
    </solution>
  </example>
