<subsection xml:id="subsec-Area-of-trapeziam">
  <title>Area of trapeziam</title>
  <p>
    Learning point 
  </p>
  <p>
    <ul>
      <li>
        <p>
          The area of a trapezium <m>= \frac{1}{2} (a+b)h</m>
        </p>
        <p>
          where: <m>(a+b)</m> is the sum of the parallel sides and <m>h</m> is the perpendicular 
          height of the trapezium.
        </p>
      </li>
    </ul>
  </p>
  <activity>
    <statement>
      <p>
        Harriate's plot is in the shape of the figure below.
      </p>
      <image source="/Trapezium.png" width="40%">
        <shortdescription>trapezium</shortdescription>
      </image>
      <p>
        What is the area of the plot of land?
      </p>
    </statement>
    <solution>
      <p>
        <md>
          <mrow> \text{Area of trapezium}=\amp \frac{1}{2} (a+b)h</mrow>
          <mrow> =\amp\frac{1}{2} (160m+200m) \times 50m</mrow>
          <mrow> =\amp \frac{1}{2} \times 360m \times 50m </mrow>
          <mrow> =\amp 9000m^2</mrow>
        </md>
      </p>
    </solution>
  </activity>
  <example>
    <statement>
      <p>
        Calculate the area of a trapezium whose parallel sides are <m>9cm</m> 
        and <m>11cm</m> and the vertical height is <m>5cm</m>
      </p>
    </statement>
    <solution>
      <p>
        <md>
          <mrow> \text{Area of a trapezium}=\amp \frac{1}{2}(9+11)cm \times 5cm </mrow>
          <mrow> =\amp \frac{1}{2} \times 20cm \times 5cm</mrow>
          <mrow> =\amp 50cm^2</mrow>
        </md>
      </p>
    </solution>
  </example>
