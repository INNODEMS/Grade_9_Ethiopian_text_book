  </p>

      <mrow> y^2=\amp 64 cm^2 +36 cm^2 </mrow>
      <mrow> y^2=\amp 100 cm^2 </mrow>
      <mrow> y=\amp \sqrt{100 cm^2} </mrow>
      <mrow> y=\amp 10 cm </mrow>
    </md>
  </p>

  </p>
