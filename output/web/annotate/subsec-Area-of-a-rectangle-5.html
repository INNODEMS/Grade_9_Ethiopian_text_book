<example>
  <statement>
    <p>
      Find the area of a rectangle whose length is <m>12 cm</m> 
      and width is <m>8cm</m>.
    </p>
  </statement>
  <hint>
    <p>
      Use Area <m>= \text{length} \times \text{width}</m>
    </p>
  </hint>
  <solution>
    <p>
      <md>
        <mrow> \text{Area of a rectangle}=\amp \text{length}\times \text{width} </mrow>
        <mrow>= \amp 12cm \times 8cm</mrow>
        <mrow> =\amp 96 cm^2 </mrow>
      </md>
    </p>
  </solution>
</example>
