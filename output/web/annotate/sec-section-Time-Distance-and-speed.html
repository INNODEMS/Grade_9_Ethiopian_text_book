      <example>
        <statement>
          <p>
            Convert <m>3</m> hours
          </p>
        </statement>
        <hint>
          <p>
            Conversion of hours to minutes, we use the 
            conversion units that is <m>1 \, hour \,= \,60 \,minuites </m>.
          </p>
        </hint>
        <solution>
          <p>
            <md>
              <mrow> 1 \, hour =\amp 60 \, minutes</mrow>
              <mrow> 3 \, hours =\amp (3 \times 60)\, minutes </mrow>
              <mrow> =\amp 180 \, minutes </mrow>
            </md>
          </p>
          <p>
            Therefore,
          </p>
          <p>
            <m>3\, hours = 180 \, minutes</m>
          </p>
          
        </solution>
      </example>
      <example>
        <statement>
          <p>
            How many minuites are the in <m>3\frac{1}{2}</m> hours?
          </p>
        </statement>
        <solution>
          <p>
            <m>1 \, hour = 60 \, minutes</m>
          </p>
          <p>
            <md>
              <mrow> 3\frac{1}{2}\,hours=\amp (3\frac{1}{2} \times 60) \, minutes</mrow>
              <mrow> =\amp (\frac{7}{2} \times 60) \, minutes</mrow>
              <mrow> =\amp \frac{420}{2} \,minuites</mrow>
              <mrow> =\amp 210 \, minutes</mrow>
            </md>
          </p>
          <p>
            Therefore,
          </p>
          <p>
            <m>3\frac{1}{2}\,hours= 210 \, minutes </m>
          </p>
        </solution>
      </example>
