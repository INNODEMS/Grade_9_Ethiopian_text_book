<example>
  <statement>
    <p>
      In the triangle find the size of line <m> PQ </m>
    </p>
    <image width="40%">
    <latex-image>
      \begin{tikzpicture}
        \draw [thick,fill=white](0,0) coordinate (A)-- (0,5) coordinate (B)-- (5,0) coordinate (C)-- (0,0)coordinate (A);
        \node at (0,-0.5){R};
        \node at (0,5.5){P};
        \node at (5,-0.5){Q};
        \node at (-1,2.5){18cm};
        \node at (3,2.5){};
        \node at (2.5,-0.5){24 cm};
        \draw(0,0) coordinate (A)-- (0,0.5) coordinate (B)-- (0.5,0.5) coordinate (C)-- (0.5,0) coordinate(D)-- (0,0) coordinate (A);
     \end{tikzpicture}
    </latex-image>
    </image>
  </statement>

  <solution>
    <p>
      <md>
        <mrow> PQ^2=\amp (QR^2+PR^2)cm </mrow>
        <mrow> PQ^2=\amp (24^2 + 18^2)cm^2</mrow>
        <mrow> PQ^2=\amp 576cm^2 + 324cm^2</mrow>
        <mrow> PQ^2=\amp \sqrt{900cm^2} </mrow>
        <mrow> PQ=\amp 30cm </mrow>
      </md>

    </p>
    <p>
      Therefore the length of <m>PQ</m> is <m>30cm</m>
    </p>
  </solution>
</example>
