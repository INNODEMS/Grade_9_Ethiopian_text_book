  <solution>
    <p>
      <md>
        <mrow> \frac{1}{2}(a+b)h=\amp \text{Area of a trapezium} </mrow>
        <mrow> \frac{1}{2}(48m+32m)h=\amp 400m^2 </mrow>
        <mrow>\frac{1}{2}(80)h= \amp 400m^2  </mrow>
        <mrow> 40h=\amp 400m^2 </mrow>
        <mrow> h=\amp\frac{400m^2 }{40} </mrow>
        <mrow> h=\amp 10m</mrow>
      </md>
    </p>
