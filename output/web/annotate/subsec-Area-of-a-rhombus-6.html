<activity>
  <statement>
    <p>
      Calculate the area of the figure given it's 
      diagonals are <m>6cm</m> and <m>8cm</m>.
    </p>
    <image source="/Rhombus.png" width="40%">
      <shortdescription>Rhombus</shortdescription>
    </image>
  </statement>
  <solution>
    <p>
      Area of a rhombus <m>= \frac{1}{2}</m> short diagonals <m>\times</m> long diagonal.
    </p>
    <p>
      <md>
        <mrow> A=\amp \frac{1}{2}d_1 d_2</mrow>
        <mrow> A=\amp \frac{1}{2} (6 \times 8)cm^2 </mrow>
        <mrow> A=\amp 24 cm^2 </mrow>
      </md>
    </p>
  </solution>
</activity>
