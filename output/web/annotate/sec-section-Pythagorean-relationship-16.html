<example>
  <statement>
    <p>
      In the triangle find the size of line <m> PQ </m>
    </p>
    <image source="/Example-2.png" width="30%">
      <shortdescription>example 3</shortdescription>
    </image>
  </statement>

  <solution>
    <p>
      <md>
        <mrow> PQ^2=\amp (QR^2-PR^2)cm </mrow>
        <mrow> PQ^2=\amp (18^2 + 24^2)cm^2</mrow>
        <mrow> PQ^2=\amp 324cm^2 + 576cm^2</mrow>
        <mrow> PQ^2=\amp \sqrt{900cm^2} </mrow>
        <mrow> PQ=\amp 30cm </mrow>
      </md>

    </p>
    <p>
      Therefore the length of <m>PQ</m> is <m>30cm</m>
    </p>
  </solution>
</example>
