<activity>
  <statement>
    <p>
      Find the area of the circle below.
    </p>
    <image width="40%">
    <latex-image>
      \begin{tikzpicture}
        \draw[very thick,fill=green](0,0) circle(5);
        \draw[thick,fill=black](0,0) -- (5,0);
        \node [very thick]at (1,0.5){$14cm$};
        \node at (0,0){.};
      \end{tikzpicture}
    </latex-image>
      
    </image>
  </statement>
  <solution>
    <p>
      <md>
        <mrow> \text{Area of a circle}=\amp \pi r^2 </mrow>
        <mrow> =\amp\frac{22}{7} \times r \times r </mrow>
        <mrow> =\amp \frac{22}{7} \times 14 \times 14 cm^2 </mrow>
        <mrow> =\amp \frac{22}{7} \times 196 cm^2</mrow>
        <mrow> =\amp 616 cm^2</mrow>
      </md>
    </p>
  </solution>
</activity>
