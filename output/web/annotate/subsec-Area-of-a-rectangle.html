<subsection xml:id="subsec-Area-of-a-rectangle">
  <title>Area of a rectangle</title>
  <p>
    Area of a rectangle <m>=</m> length <m>\times</m> width
  </p>
  <p>
    length <m>=</m> Area <m>\div</m> width
  </p>
  <p>
    width <m>=</m> Area <m>\div</m> length
  </p>
  <example>
    <statement>
      <p>
        Find the area of a rectangle whose length is <m>12 cm</m> 
        and width is <m>8cm</m>.
      </p>
    </statement>
    <hint>
      <p>
        Use Area <m>= \text{length} \times \text{width}</m>
      </p>
    </hint>
    <solution>
      <p>
        <md>
          <mrow> \text{Area of a rectangle}=\amp \text{length}\times \text{width} </mrow>
          <mrow>= \amp 12cm \times 8cm</mrow>
          <mrow> =\amp 96 cm^2 </mrow>
        </md>
      </p>
    </solution>
  </example>
  <activity>
    <statement>
      <p>
        A rectangular plot is of length <m>300m</m> 
        and width <m>100m</m>. Find its area in;
      </p>
      <p>
        <ol>
          <li>
            <p>
              <m>m^2</m>
            </p>
          </li>
          <li>
            <p>
              <m>ha</m>
            </p>
          </li>
        </ol>
      </p>
    </statement>
    <solution>
      <p>
        1.Find the area in <m>m^2</m> that is
      </p>
      <p>
        <md>
          <mrow>\text{Area}= \amp \text{length}\times \text{width}</mrow>
          <mrow> =\amp 300m \times 100m</mrow>
          <mrow> =\amp 30000m^2</mrow>
        </md>
      </p>
      <p>
        2. Converting to <m>ha</m>
      </p>
      <p>
        <md>
          <mrow> \text{if} 1 \text{ha}=\amp 10000m^2</mrow>
          <mrow> 30000m^2=\amp \frac{30000m^2 \times 1 ha }{10000m^2} </mrow>
          <mrow> =\amp 3ha</mrow>
        </md>
      </p>
    </solution>
  </activity>
