<subsection xml:id="subsec-Multiplication-involving-units-of-length">
  <title>Multiplication involving units of length</title>
  <activity>
    <statement>
      <p>
        Prepare a number of sticks of the same length
      </p>
      <p>
        place them end to end.
      </p>
      <p>
        Measure their total length.
      </p>
      <p>
        Relate the length of each stick to the total length.
      </p>
    </statement>
  </activity>
  <example>
    <statement>
        <p>
        Work out <m>8\text{ Dm }</m> <m>9\text{ m } \times3</m>
        </p>
    </statement>
    <solution>
      <sidebyside>
        
        <tabular>

            <cell>m</cell>

            <cell>9</cell>
          </row>

            <cell> <m>\times</m> </cell>
            <cell>3</cell>
          </row>
          <row bottom="medium">
            <cell>26</cell>
            <cell>7</cell>
          </row>
        </tabular>
       
       <p>
        <ul>
          <li>
            <p>
              <m>\rightarrow</m> <m>9 \times3 = 27</m>, write <m>7\text{ m } </m> regroup <m>20 \text{ m }</m> to <m>2\text{ Dm }.</m>
            </p>
          </li>
          <li>
            <p>
               <m>\rightarrow</m> <m>8\times3 =24 +2 = 26</m>

          </li>
         </ul>
         </p>
      </sidebyside>
       
    </solution>
  </example>

  <activity>
    <statement>
      <p>
        <ul>
          <li>
            <p>
              Prepare a string of a certain length.
            </p>
          </li>
          <li>
            <p>
              Fold and cut it into pieces of equal length.
            </p>
          </li>
          <li>
            <p>
              Measure the length of each piece of string.
            </p>
          </li>
          <li>
            <p>
              Relate the length of each string to tha of the original string.
            </p>
          </li>
        </ul>
        Prepare
      </p>
    </statement>
  </activity>
  <example>
    <statement>
      <p>
        Work out <m>13 \text{ dm } </m> <m>2 \text{ m } \div 6</m>
      </p>
    </statement>
    <solution>
      <sidebyside>
        
          
          <tabular>
            <row>
              <cell />
              <cell bottom="medium">2Dm</cell>
              <cell bottom="medium">2m</cell>
            </row>
            <row>
              <cell right="medium">6</cell>

              <cell>2m</cell>
            </row>
            <row>
              <cell />

              <cell />
            </row>
            <row>

              <cell>10</cell>
            </row>
            <row>
              <cell />
              <cell />
              <cell bottom="medium">+2</cell>
            </row>
            <row>
              <cell />
              <cell />
              <cell>12</cell>
            </row>
            <row>
              <cell />
              <cell />
              <cell bottom="medium">-12</cell>
            </row>
            <row>
              <cell />
              <cell />
              <cell bottom="medium">00</cell>
            </row>
          </tabular>
        
        <p>
          <ul>
            <li>
              <p>
                Divide <m>13\text{ Dm }</m> by <m>6</m> to get <m>2\text{ Dm }</m> remainder <m>1\text{ Dm }</m>.
              </p>
            </li>
            <li>
              <p>
                Convert <m>1\text{ Dm }</m> to <m>\text{ m }</m> to get <m>10\text{ m }</m>.
              </p>
            </li>
            <li>
              <p>
                Add <m>2\text{ m }</m> to <m>10\text{ m }</m> to get <m>12\text{ m }</m>.
              </p>
            </li>
            <li>
              <p>
                Divide <m>12\text{ m }</m> by <m>6</m> to get <m>2\text{ m }</m>.
              </p>
            </li>
            </ul>
            <m>13 \text{ dm } </m> <m>2 \text{ m } \div 6 = 2 \text{ Dm } 2 \text{ m }</m> 
        </p>
      </sidebyside>
    </solution>
