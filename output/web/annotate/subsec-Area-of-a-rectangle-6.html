<activity>
  <statement>
    <p>
      A rectangular plot is of length <m>300m</m> 
      and width <m>100m</m>. Find its area in;
    </p>
    <p>
      <ol>
        <li>
          <p>
            <m>m^2</m>
          </p>
        </li>
        <li>
          <p>
            <m>ha</m>
          </p>
        </li>
      </ol>
    </p>
  </statement>
  <solution>
    <p>
      1.Find the area in <m>m^2</m> that is
    </p>
    <p>
      <md>
        <mrow>\text{Area}= \amp \text{length}\times \text{width}</mrow>
        <mrow> =\amp 300m \times 100m</mrow>
        <mrow> =\amp 30000m^2</mrow>
      </md>
    </p>
    <p>
      2. Converting to <m>ha</m>
    </p>
    <p>
      <md>
        <mrow> \text{if} 1 \text{ha}=\amp 10000m^2</mrow>
        <mrow> 30000m^2=\amp \frac{30000m^2 \times 1 ha }{10000m^2} </mrow>
        <mrow> =\amp 3ha</mrow>
      </md>
    </p>
  </solution>
</activity>
