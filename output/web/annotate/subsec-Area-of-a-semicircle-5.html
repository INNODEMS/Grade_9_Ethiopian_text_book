<activity>
  <statement>
    <p>
      Find the area of the semicircle given below.
    </p>
    <image width="40%">
    <latex-image>
      \begin{tikzpicture}
      \draw[very thick,fill=white](0,0)  arc[start angle=0,end angle=180,radius=5cm];
      \draw [thick,fill=black](0,0) --(-10,0);
      \node at (-5,0.5){$28cm$};
      
      \end{tikzpicture}
    </latex-image>
      
    </image>
  </statement>
  <solution>
    <p>
      <md>
        <mrow> \text{radius} =\amp \frac{\text{diameter}}{2} </mrow>
        <mrow> r=\amp \frac{28}{2}cm </mrow>
        <mrow> r=\amp 14cm </mrow>
      </md>
    </p>
    <p>
      <md>
        <mrow> \text{Area of semicircle}=\amp \frac{1}{2} \pi \times r^2 </mrow>
        <mrow> =\amp \frac{1}{2} \pi \times r \times r</mrow>
        <mrow> =\amp\frac{1}{2}\times\frac{22}{7} \times 14^2cm^2 </mrow>
        <mrow> =\amp \frac{1}{2} \times\frac{22}{7} \times 196 cm^2</mrow>
        <mrow> =\amp\frac{1}{2} \times 616cm^2 </mrow>
        <mrow> =\amp 308cm^2</mrow>
      </md>
      
    </p>
  </solution>
</activity>
