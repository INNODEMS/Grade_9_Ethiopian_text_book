<example>
  <statement>
    <p>
      Find the length of <m> w </m> in the figure below.
    </p>
    <image width="50%">
    <latex-image>
      \begin{tikzpicture}
        \draw [thick,fill=blue] (0,4) --(4,4) --(4,0) --(0,4);
        \node at (2,1){$40 \,cm$};
        \node at (2,4.5){$w \,cm$};
        \node at (4.5,2){$24 \,cm$};
        \draw (3.5,3.5) --(3.5,4) --(4,4) --(4,3.5) --(3.5,3.5);
      \end{tikzpicture}
    </latex-image>
      
    </image>
  </statement>
  <solution>
    <p>
      <md>
        <mrow> w^2=\amp (40^2-24^2)cm </mrow>
        <mrow> w^2=\amp 1 600cm^2 - 576cm^2</mrow>
        <mrow> w^2=\amp 1 024cm^2</mrow>
        <mrow> w=\amp \sqrt{1024cm^2} </mrow>
        <mrow> w=\amp 32cm </mrow>
      </md>
    </p>
    <p>
      Therefore the length of <m>w</m> is <m>32cm</m>
    </p>
  </solution>
</example>
