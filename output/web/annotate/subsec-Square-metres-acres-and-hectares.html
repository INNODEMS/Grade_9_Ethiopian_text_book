  <subsubsection xml:id="subsubsec-Hectares">
    <title>Hectares</title>
    <p>
      A hectare is a unit of measuring large areas.
    </p>
    <p>
      One hectare is equal to <m>10000 m^2</m>
    </p>
    <p>
      Therefore,
    </p>
    <p>
      <m>1 \text{hectare}(ha) = 10000m^2</m>
    </p>
    <p>
      <m>1 \text{hectare}(ha) = 2.47 \text{acre}</m>
    </p>
    <example>
    <statement>
      <p>
        Convert <m>25 \text{ha}</m> into <m>\text{ares}</m>
      </p>
    </statement>
    <hint>
      <p>
        Use <m>1 hectare(ha) = 2.47 acre</m>
      </p>
    </hint>
    <solution>
      <p>
        <md>
          <mrow>  1ha=\amp 2.47 \text{acre} </mrow>
          <mrow> 25 ha=\amp (2.47 \times 25) \text{acres}</mrow>
          <mrow> =\amp 61.75 acres</mrow>
        </md>
      </p>
    </solution>
    </example>
    <example>
      <statement>
        <p>
          find the area of the figure below in hectares.
        </p>
        <image width="30%">
        <latex-image>
          \begin{tikzpicture}
            \draw(0,0) coordinate (A)-- (0,4) coordinate (B)-- (4,4) coordinate (C)-- (4,0)coordinate (D) --(0,0)coordinate (A);
            \node at (4.5,2){600 m};
            \node at (2,-0.5){600 m};
            \draw(0,0) coordinate (A)-- (0,0.5) coordinate (B)-- (0.5,0.5) coordinate (C)-- (0.5,0)
            coordinate(D)-- (0,0) coordinate (A);
            \draw(0,3.5) coordinate (A)-- (0,4) coordinate (B)-- (0.5,4) coordinate (C)-- (0.5,3.5)
            coordinate(D)-- (0,3.5) coordinate (A);
            \draw(3.5,3.5) coordinate (A)-- (3.5,4) coordinate (B)-- (4,4) coordinate (C)-- (4,3.5)
            coordinate(D)-- (3.5,3.5) coordinate (A);
            \draw(3.5,0) coordinate (A)-- (3.5,0.5) coordinate (B)-- (4,0.5) coordinate (C)-- (4,0)
            coordinate(D)-- (3.5,0) coordinate (A);
          \end{tikzpicture}
        </latex-image>
            
        </image>

      </statement>
      <solution>
        <p>
          Find the area of a square.
        </p>
        <p>
          Area of a square <md>
          <mrow> =\amp \text{side} \times \text{side}  </mrow>
          <mrow> =\amp 600m \times 600m</mrow>
          <mrow> =\amp 360000m^2</mrow>
          </md>

        </p>
        <p>
          Convert <m>360000 m^2</m> to hectares.
        </p>
        <p>
          <md>
            <mrow> 10 000m^2=\amp 1 ha</mrow>
            <mrow> 360000 m^2=\amp (360000 \div 10000) \text{hectares} </mrow>
            <mrow> =\amp 36 \text{hectares}</mrow>
          </md>

        </p>
        <p>
          The area of the figure is <m>36 </m> hectares
        </p>
      </solution>



    </example>
