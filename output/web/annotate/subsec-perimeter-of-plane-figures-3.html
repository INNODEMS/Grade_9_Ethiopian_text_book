  <solution>
    <p>
      <ol cols="2" marker="a)">
        <li>
          <p> Perimeter of a square.
            <md>
              <mrow>=\amp l+l+l+l </mrow>
              <mrow>=\amp 4l</mrow>
              <mrow>=\amp 4 \times 10cm </mrow>
              <mrow>=\amp  \textbf {40cm} </mrow>
            </md>
          </p>
        </li>
        <li>
            <p> Perimeter of a rectangle.
            <md>
              <mrow>=\amp l+w+l+w </mrow>
              <mrow>=\amp 2 \times (l + w) </mrow>
              <mrow>=\amp 2 \times (40cm +8cm) </mrow>
              <mrow>=\amp 2 \times 48cm </mrow>
              <mrow>=\amp  \textbf {96cm} </mrow>

            </md>
          </p>
        </li>
      </ol>
    </p>
