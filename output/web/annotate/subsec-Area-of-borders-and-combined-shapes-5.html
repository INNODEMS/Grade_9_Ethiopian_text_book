<activity>
  <statement>
    <p>
      Work out the area of the figure bellow
    </p>
    <image width="50%">
    <latex-image>
      \begin{tikzpicture}
        \draw[very thick,fill=white] (0,0) --(5,0) --(8,0) --(5,3) --(0,3);
        \draw [very thick,fill=white]   (0,3) arc[start angle=90,end angle=270,radius=1.5cm] ;
        \draw [dashed,fill=black] (0,0) --(0,3);
        \draw [dashed,fill=black] (5,0) --(5,3);
        \draw [very thick, &lt;-&gt;] (0,-0.5) --(5,-0.5);
        \draw [very thick, &lt;-&gt;] (5,-0.5) --(8,-0.5) ;
        \node at (2.5,-1){$20cm$};
        \node at (6,-1){$18cm$};
        \node at (0.5,1.5){$14cm$};
      \end{tikzpicture}
    </latex-image>
      
    </image>
  </statement>
  <solution>
   <p>
      <md>
        <mrow> \text{Area of the rectangle}=\amp l \times w </mrow>
        <mrow> =\amp 20cm \times 14cm </mrow>
        <mrow> =\amp280 cm^2 </mrow>
      </md>
   </p>
   <p>
      <md>
        <mrow> \text{Area of the triangle}=\amp \frac{1}{2} bh </mrow>
        <mrow> =\amp  \frac{1}{2} \times 18cm \times 14cm  </mrow>
        <mrow>= \amp 126cm^2</mrow>
      </md>
    </p>
   <p>
      <md>
        <mrow> \text{Area of the semicircle}=\amp \frac{1}{2} \pi r^2 </mrow>
        <mrow> =\amp \frac{1}{2} \frac{22}{7} \times 7cm \times 7cm</mrow>
        <mrow> =\amp77cm^2 </mrow>
      </md>
   </p>
   <p>
      <md>
        <mrow> \text{Area of the combined shapes}=\amp \text{area of all the three figures calculated} </mrow>
        <mrow> =\amp 280cm^2 +126cm^2+77cm^2 </mrow>
        <mrow> =\amp 483cm^2</mrow>
      </md>
   </p>
   <p>
     <m>Alternatively:</m> add up the area of the trapezium and the semicircle.
   </p>
   <p>
      <md>
        <mrow> \text{Area of the trapezium}=\amp \frac{1}{2} h(a+b)  </mrow>
          <mrow> =\amp  \frac{1}{2} \times 14(20+18+20) </mrow>
          <mrow> = \amp 7cm \times 58cm </mrow>
          <mrow> = \amp406cm^2 </mrow>
      </md>
   </p>
   <p>
     Area of the semi circle <m>=77cm^2</m>
   </p>
   <p>
      <md>
        <mrow>\text{Area of the combined shape} =\amp \text{Area of the two figure} </mrow>
        <mrow> = \amp 406cm^2+77cm^2 </mrow>
        <mrow> =\amp 483cm^2 </mrow>
      </md>
   </p>

    
  </solution>
</activity>
